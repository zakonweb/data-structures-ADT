Linked List Addition Conceputual Pseudocode:

Input Data (D)
If FREE = Null 
  Then
    Output "Overflow"
    End
End If

//Borrow a node from free list
//And adding data to it.
TempFree = FREE
FREE = FREE.Next
TempFree.Data = D

If List1 = Null 
  Then
    List1 = TempFree
    TempFree.Next = Null
    End
End If

If List1 <> Null
  Then
    CN = List1
    If (CN.Data < D) 
      Then
	While (CN.Next.Data < D) AND (CN.Next <> Null)
            CN = CN.Next
        End While
    End If
    Temp2 = List1
    List1 = TempFree
    TempFree.Next = Temp2
End If
